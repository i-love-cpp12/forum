// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs
Table user_role {
  user_role_id integer [primary key]
  role_name varchar(50) [not null]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: null]
}

Table _user {
  user_id integer [primary key]
  email varchar(320) [not null, unique]
  username varchar(50) [not null]
  password_hash varchar(64) [not null]
  user_role_id integer [default: 1, not null]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: null]
}

Table user_token {
  user_token_id integer [primary key]
  user_id integer [not null]
  value varchar(64) [not null, unique]
  created_at timestamp [default: `now()`, not null]
  expire_at timestamp
}

Table like_type {
  like_type_id integer [primary key]
  like_type_name varchar(50) [not null]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: null]
}

Table _like {
  like_id integer [primary key]
  post_id integer [not null]
  user_id integer [not null]
  like_type_id integer [not null]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: null]
}

Table post_category {
  post_category_id integer [primary key]
  post_category_name varchar(100) [not null]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: null]
}
Table post_post_category {
  post_post_category integer [primary key]
  post_id integer [not null]
  post_category_id integer [not null]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: null]
}
Table post {
  post_id integer [primary key]
  parent_post_id integer
  user_id integer [not null]
  header varchar(100)
  content varchar(1000) [not null]
  like_count integer [default: 0, not null]
  dislike_count integer [default: 0, not null]
  comment_count integer [default: 0, not null]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: null]
}

Ref _user_user_role: _user.user_role_id > user_role.user_role_id

Ref _user_token_user: user_token.user_id > _user.user_id

Ref _like_post: _like.post_id > post.post_id
Ref _like_like_type: _like.like_type_id > like_type.like_type_id
Ref _like_user: _like.user_id > _user.user_id

Ref post_user: post.user_id > _user.user_id
Ref post_post: post.parent_post_id > post.post_id

Ref post_post_category_post: post_post_category.post_id > post.post_id
Ref post_post_category_post_category: post_post_category.post_category_id > post_category.post_category_id
